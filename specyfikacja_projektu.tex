	\section{Tytuł Projektu}
	System wykrywania anomalii w ruchu sieciowym (NIDS) w Javie
	
	\section{Opis i Cele Projektu}
	Projekt ma na celu stworzenie aplikacji monitorującej ruch sieciowy w czasie rzeczywistym, która analizuje pakiety i wykrywa potencjalne zagrożenia bezpieczeństwa.
	
	Aplikacja umożliwia:
	\begin{itemize}
		\item przechwytywanie i analizę pakietów sieciowych,
		\item wykrywanie podejrzanych aktywności,
		\item generowanie alertów i raportów bezpieczeństwa.
	\end{itemize}
	
	Główne cele projektu:
	\begin{itemize}
		\item identyfikacja anomalii,
		\item analiza statystyczna ruchu,
		\item wizualizacja zdarzeń,
		\item powiadamianie o zagrożeniach.
	\end{itemize}
	
	\section{Wymagania Funkcjonalne}
	\begin{itemize}
		\item Przechwytywanie pakietów (TCP, UDP, ICMP)
		\item Analiza nagłówków pakietów
		\item Wykrywanie nadmiernej liczby połączeń
		\item Wykrywanie skanowania portów
		\item System alertów
		\item Generowanie raportów
		\item Filtrowanie zdarzeń
		\item Konfiguracja progów detekcji
	\end{itemize}
	
	\section{Wymagania Niefunkcjonalne}
	\begin{itemize}
		\item Wydajność: min. 100 pakietów/sek
		\item Skalowalność: obsługa wielu interfejsów
		\item Niezawodność: odporność na błędy
		\item Użyteczność: prosty interfejs
	\end{itemize}
	
	\section{Architektura Systemu}
	Typ architektury: warstwowa
	
	Komponenty:
	\begin{itemize}
		\item Capture Layer – przechwytywanie pakietów
		\item Analysis Layer – analiza danych
		\item Detection Engine – logika detekcji
		\item Alert System – powiadomienia
		\item Storage – zapis danych
	\end{itemize}
	
	\section{Zastosowane Algorytmy}
	\begin{itemize}
		\item Detekcja progowa (threshold-based)
		\item Analiza anomalii
		\item Parsowanie pakietów TCP/IP
		\item Filtrowanie strumieniowe
	\end{itemize}
	
	\section{Struktura Klas}
	\subsection*{Model}
	\begin{itemize}
		\item Packet - dane pakietu (IP, porty, protokół)
		\item Alert - wykryte zdarzenie
		\item TrafficStats - statystyki ruchu
	\end{itemize}
	
	\subsection*{Serwisy}
	\begin{itemize}
		\item PacketCaptureService - przechwytywanie ruchu
		\item PacketAnalysisService - analiza pakietów
		\item DetectionService - logika wykrywania
		\item AlertService - generowanie alertów
	\end{itemize}
	
	\subsection*{Kontrolery}
	\begin{itemize}
		\item AlertController
		\item StatsController
	\end{itemize}
